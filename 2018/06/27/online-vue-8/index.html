路由懒加载 8 | Vue

作者： Lian

日期： 2018年06月27日 10:06

1 修改工程目录结构

  vue1
  |- package.json
+ |- index.html
+ |- .babelrc
+ |- /src
+   |- /page
+     |- HelloWorld.vue
+   |- /router
+     |- index.js
+   |- /components
+     |- HelloWorld.vue
+   |- /util
+     |- store.js

+   |- main.js
+   |- style.css
+ |- /build
+   |- webpack.base.js
+   |- webpack.dev.js
+   |- webpack.prod.js

2 安装依赖

npm install --save-dev babel-plugin-syntax-dynamic-import

3 修改文件 .babelrc

{
  "presets": [ "es2015" ],
  "plugins": ["syntax-dynamic-import"]
}

4 修改文件 \src\router\index.js

import Vue from 'vue'
import Router from 'vue-router'

// 路由推荐使用懒加载，提升SPA加载性能。
const HelloWorld = () => import('../page/HelloWorld.vue');
const Error = () => import('../page/Error.vue');

Vue.use(Router)

export default new Router({
  routes: [
    {
      path: '/',
      name: 'HelloWorld',
      component: HelloWorld
    },
    {
      path: '/error',
      name: 'Error',
      component: Error
    }
  ]
})
